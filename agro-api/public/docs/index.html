    const url = new URL(
    "http://localhost:8082/api/auth/login"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "email": "a",
    "senha": "est"
}

fetch(url, {
    method: "POST",
    headers: headers,
    body: body
